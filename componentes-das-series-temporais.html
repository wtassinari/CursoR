5.1 Estacionariedade

  • A média e a variância se mantêm constante durante o tempo

  • Exemplo:: Incidência de dengue nas Filipinas, 2008 - 2016
d=read.csv("/home/tassinari/Documentos/cursos_ecologicos_2019/dados/denguecases.csv")

DT::datatable(d, filter="top", options = list(pageLength = 5, scrollX=T))
# Agregando os dados regionais para representar todo o país
denguecasesph=aggregate(Dengue_Cases~Month+Year,d,sum)

#Convertendo os dados para o formato de Séries Temporais
denguecasesphts=ts(denguecasesph$Dengue_Cases,c(2008,1),c(2016,12),12)

#12 foi especificado pois queremos mostrar dos dados mensais
plot(denguecasesphts)
library(Kendall)
lines(lowess(time(denguecasesphts),denguecasesphts),lwd=3, col=2)

library(ggfortify)
# Decomposição da Série Temporal
autoplot(decompose(denguecasesphts))
